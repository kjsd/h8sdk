 160 static char* test_sci()
 161 {
 162     const char str[] = "ABCabc\r\n";
 163     unsigned char buf[5] = { 0 };
 164 
 165     SCI_init();
 166     UTEST_assert("test_sci: SMR != 0", _SCI3.SMR.BYTE == 0);
 167     UTEST_assert("test_sci: BRR != 0x20", _SCI3.BRR == 0x20);
 168     UTEST_assert("test_sci: PMR1 != 1", _IO.PMR1.BIT.TXD == 1);
 169     UTEST_assert("test_sci: SCR3 != 0x70", _SCI3.SCR3.BYTE == 0x70);
 170 
 171     UTEST_assert("test_sci: SCI_puts", SCI_puts(str) == (sizeof(str) - 1));
 172     UTEST_assert("test_sci: SCI_write(1)",
 173                  SCI_write((_UBYTE*)str, sizeof(str) - 1, _TRUE, -1) ==
 174                  (sizeof(str) - 1));
 175 
 176     SCI_puts("test_sci: Reading serial port... Press 'q' to quit.\r\n");
 177     while (buf[0] != 'q')
 178     {
 179         if (SCI_read(buf, 1, _FALSE, -1) == 1)
 180         {
 181             SCI_write(buf, 1, _FALSE, -1);
 182         }
 183     }
 184     SCI_puts("\r\n");
 185 
 186     _printf("test_sci: ALL OK.\r\n");
 187     return 0;
 188 }

 263     UTEST_assert("test_stdlib: _next_ring(2)", _next_ring(0, 0) == 0);

 271     UTEST_assert("test_stdlib: _hash(4)", _hash(3, 5) == 3);
 272     UTEST_assert("test_stdlib: _hash(5)", _hash(4, 5) == 4);
 273 
 274     UTEST_assert("test_stdlib: _check_sum(1)",
 275                  _check_sum(ary, sizeof(ary) / sizeof(_UWORD)) == 0x0202U);
 276

 287

 289     UTEST_assert("test_stdlib: _itoa(6)", num == 2);
 290     str[num] = '\0';
 291     UTEST_assert("test_stdlib: _itoa(7)", str[0] == 0x31);
 292     UTEST_assert("test_stdlib: _itoa(8)", str[1] == 0x32);

 340 #ifdef TEST_LCD
 341 static char* test_lcd()
 342 {
 343     int i;
 344     unsigned char buf[LCD_MAX_COLS * LCD_MAX_ROWS];
 345     int x, y;
 346     unsigned char ch[LCD_CG_ROWS] = { 0 };
 347 
 348     LCD_init();
 349     UTEST_assert("test_lcd: LCD_init(1)",
 350                  LCD_read(buf, sizeof(buf)) == sizeof(buf));
 351     for (i = 0; i < sizeof(buf); i++)
 352     {
 353         UTEST_assert("test_lcd: LCD_init(2)", buf[i] == ' ');
 354     }
 355 
 356     LCD_setCursor(0, 0);
 357     LCD_getCursor(&x, &y);
 358     UTEST_assert("test_lcd: LCD_(set|get)Cursor(1)", (x == 0) && (y == 0));
 359     LCD_setCursor(LCD_MAX_COLS, LCD_MAX_ROWS);
 360     LCD_getCursor(&x, &y);
 361     UTEST_assert("test_lcd: LCD_(set|get)Cursor(2)", (x == 0) && (y == 0));
 362 
 363     LCD_setCursor(0, 0);
 364     LCD_puts("aaa\n");
 365     LCD_getCursor(&x, &y);
 366     UTEST_assert("test_lcd: LCD_(set|get)Cursor(3)", (x == 0) && (y == 1));
 367 
 368     LCD_setCursor(0, 0);
 369     UTEST_assert("test_lcd: LCD_puts(1)", LCD_read(buf, 3) == 3);
 370     UTEST_assert("test_lcd: LCD_puts(2)",
 371                  (buf[0] == 'a') && (buf[1] == 'a') && (buf[2] == 'a'));
 372     
 373     LCD_setCursor(0, 0);
 374     _memset(buf, 'a', sizeof(buf));
 375     UTEST_assert("test_lcd: LCD_write(1)",
 376                  LCD_write(buf, sizeof(buf)) == sizeof(buf));
 377 
 378     LCD_setCursor(0, 0);
 379     _memset(buf, 0, sizeof(buf));
 380     UTEST_assert("test_lcd: LCD_read(1)",
 381                  LCD_read(buf, sizeof(buf)) == sizeof(buf));
 382     for (i = 0; i < sizeof(buf); i++)
 383     {
 384         UTEST_assert("test_lcd: LCD_read(2)", buf[i] == 'a');
 385     }
 386 
 387     for (i = 0; i < LCD_CG_MAXCHAR; i++)
 388     {
 389         LCD_setChar(i, ch);
 390         buf[i] = i;
 391         ch[0]++;
 392     }
 393 
 394     LCD_setCursor(0, 0);
 395     UTEST_assert("test_lcd: LCD_write(1)",
 396                  LCD_write(buf, LCD_CG_MAXCHAR) == LCD_CG_MAXCHAR);
 397 
 398     _printf("test_lcd: ALL OK.\r\n");
 399     return 0;
 400 }
