 141 _SINT SSRP_send(SSRP_Header* header, const _UBYTE* data, _UBYTE len, _BOOL sync)
 142 {
 143     _SINT rsize = 0;
 144     _UBYTE pre = SSRP_VAL_PREAMBLE;
 145     _SINT i;
 146 
 147     _assert(stub_);
 148     _assert(len <= header->data_length);
 149 
 150     for (i = trx_[SSRP_TRX_SEND].state; i < NUM_OF_TRXST_; i++)
 151     {
 152         switch (i)
 153         {
 154         case TRXST_PREAMBLE_:
 155             rsize = stub_->write(&pre, sizeof(pre), sync, SSRP_TMO_SEND);
 156             if (rsize != sizeof(pre)) rsize = -1;
 157             break;
 158 
 159         case TRXST_HEADER_:
 160             if (header->from == my_addr_)
 161             {
 162                 header->trx_id = trx_count_++;
 163             }
 164             header->cksum = _check_sum(header,
 165                                        (sizeof(SSRP_Header) -
 166                                         sizeof(header->cksum)) /
 167                                        sizeof(_UWORD));
 168             header->cksum = ~(header->cksum) + 1;
 169 
 170             rsize = stub_->write((_UBYTE*)header, sizeof(SSRP_Header), _TRUE,
 171                                  SSRP_TMO_SEND);
 172             break;
 173 
 174         case TRXST_DATA_:
 175             rsize = stub_->write(data, len, sync, SSRP_TMO_SEND);
 176             break;
 177 
 178         default:
 179             _assert(_FALSE);
 180             break;
 181         }
 182 
 183         if (rsize < 0) goto func_END;
 184     }
 185     
 186     if (i > TRXST_DATA_)
 187     {
 188         trx_[SSRP_TRX_SEND].state = (rsize < len) ?
 189             TRXST_DATA_: TRXST_PREAMBLE_;
 190     }
 191     else
 192     {
 193         trx_[SSRP_TRX_SEND].state = i;
 194     }
 195 
 196 func_END:
 197     return rsize;
 198 }

 200 _SINT SSRP_sendto(SSRP_Address to, SSRP_Command cmd, SSRP_Flag flg,
 201                   const _UBYTE* data, _UBYTE len,
 202                   SSRP_TransactionId* trx_id, _BOOL sync)
 203 {
 204     _SINT rtn;
 205 
 206     trx_[SSRP_TRX_SEND].header.from = my_addr_;
 207     trx_[SSRP_TRX_SEND].header.to = to;
 208     trx_[SSRP_TRX_SEND].header.command = cmd;
 209     trx_[SSRP_TRX_SEND].header.flag = flg;
 210     trx_[SSRP_TRX_SEND].header.data_length = len;
 211 
 212     rtn = SSRP_send(&trx_[SSRP_TRX_SEND].header, data, len, sync);
 213     if (trx_id != NULL)
 214     {
 215         *trx_id = trx_[SSRP_TRX_SEND].header.trx_id;
 216     }
 217     
 218     return rtn;
 219 }

 222 {
 223     if ((trx_[SSRP_TRX_RECV].header.to != my_addr_) &&
 224         (trx_[SSRP_TRX_RECV].header.to != SSRP_ADDR_BROADCAST))
 225     {
 226         return -1;
 227     }
 228     if (trx_[SSRP_TRX_RECV].state != TRXST_PREAMBLE_)
 229     {
 230         return -1;
 231     }
 232 
 233     *header_p = &trx_[SSRP_TRX_RECV].header;
 234     *data_p = trx_[SSRP_TRX_RECV].buffer;
 235 
 236     return trx_[SSRP_TRX_RECV].header.data_length;

 245 void SSRP_exec()
 246 {
 247     _SINT rtn;
 248     _UBYTE rsize;
 249     _UBYTE buf;
 250     
 251     SSRP_INVALIDATE_PACKET(&trx_[SSRP_TRX_RECV].header);
 252 
 253     rtn = recv_(trx_[SSRP_TRX_RECV].buffer, SSRP_MAX_DATA, _FALSE);
 254     if (rtn >= 0)
 255     {
 256         for (rsize = rtn;
 257              rsize < trx_[SSRP_TRX_RECV].header.data_length;
 258              rsize += rtn)
 259         {
 260             rtn = recv_(trx_[SSRP_TRX_RECV].buffer + rsize,
 261                             trx_[SSRP_TRX_RECV].header.data_length - rsize,
 262                             _FALSE);
 263             if (rtn < 0)
 264             {
 265                 SSRP_shutdown(SSRP_TRX_RECV);
 266                 break;
 267             }
 268         }
 269     }
 270     else
 271     {
 272         SSRP_shutdown(SSRP_TRX_RECV);
 273     }
 274 
 275     if (start_)
 276     {
 277         if (state_tbl_[current_state_].handle())
 278         {
 279             current_state_ = state_tbl_[current_state_].next;
 280         }
 281     }
 282 
 283     if (SSRP_IS_INVALID_PACKET(&trx_[SSRP_TRX_RECV].header)) return;
 284 
 285     if (trx_[SSRP_TRX_RECV].header.from == my_addr_) return;
 286     if (trx_[SSRP_TRX_RECV].header.to == my_addr_) return;
 287     if (trx_[SSRP_TRX_RECV].header.hop >= pear_.total) return;
 288 
 289     if (start_)
 290     {
 291         trx_[SSRP_TRX_RECV].header.hop++;
 292     }
 293     SSRP_send(&trx_[SSRP_TRX_RECV].header,
 294               trx_[SSRP_TRX_RECV].buffer,
 295               trx_[SSRP_TRX_RECV].header.data_length,
 296               _TRUE);
 297 }

 299 static _SINT recv_(_UBYTE* buf, _UBYTE len, _BOOL sync)
 300 {
 301     _SINT rsize = 0;
 302     _UBYTE pre = 0;
 303     _SINT i;
 304 
 305     _assert(stub_);
 306     _assert(len <= SSRP_MAX_DATA);
 307 
 308     for (i = trx_[SSRP_TRX_RECV].state; i < NUM_OF_TRXST_; i++)
 309     {
 310         switch (i)
 311         {
 312         case TRXST_PREAMBLE_:
 313             rsize = stub_->read(&pre, sizeof(pre), sync, SSRP_TMO_RECV);
 314             if (pre != SSRP_VAL_PREAMBLE)
 315             {
 316                 rsize = -1;
 317             }
 318             break;
 319 
 320         case TRXST_HEADER_:
 321             rsize = stub_->read((_UBYTE*)&trx_[SSRP_TRX_RECV].header,
 322                                 sizeof(SSRP_Header), _TRUE, SSRP_TMO_RECV);
 323 
 324             if (rsize != sizeof(SSRP_Header))
 325             {
 326                 rsize = -1;
 327                 break;
 328             }
 329 
 330             if (_check_sum(&trx_[SSRP_TRX_RECV].header,
 331                            sizeof(SSRP_Header) / sizeof(_UWORD)) != 0)
 332             {
 333                 rsize = -1;
 334             }
 335             break;
 336 
 337         case TRXST_DATA_:
 338             if (len > trx_[SSRP_TRX_RECV].header.data_length)
 339             {
 340                 len = trx_[SSRP_TRX_RECV].header.data_length;
 341             }
 342             rsize = stub_->read(buf, len, sync, SSRP_TMO_RECV);
 343             break;
 344 
 345         default:
 346             _assert(_FALSE);
 347             break;
 348         }
 349 
 350         if (rsize < 0) goto func_END;
 351     }
 352 
 353     if (i > TRXST_DATA_)
 354     {
 355         trx_[SSRP_TRX_RECV].state = (rsize < len) ?
 356             TRXST_DATA_: TRXST_PREAMBLE_;
 357     }
 358     else
 359     {
 360         trx_[SSRP_TRX_RECV].state = i;
 361     }
 362 
 363 func_END:
 364     return rsize;
 365 }

 367 static _BOOL stHdlCountPear_()
 368 {
 369     static _UWORD counter = SSRP_PACKET_SPC;
 370     _UBYTE buf = 0;
 371 
 372 
 373     if (!SSRP_IS_INVALID_PACKET(&trx_[SSRP_TRX_RECV].header))
 374     {
 375         if ((trx_[SSRP_TRX_RECV].header.from == my_addr_) &&
 376             SSRP_COMMAND_EQ(&trx_[SSRP_TRX_RECV].header,
 377                             SSRP_CMD_LOOP, SSRP_FLG_EVT))
 378         {
 379             pear_.total = trx_[SSRP_TRX_RECV].header.hop;
 380 
 381             return _TRUE;
 382         }
 383     }
 384 
 385     if (counter++ & SSRP_PACKET_SPC)
 386     {
 387         SSRP_sendto(SSRP_ADDR_BROADCAST, SSRP_CMD_LOOP, SSRP_FLG_EVT,
 388                     &buf, sizeof(buf), NULL, _TRUE);
 389     }
 390 
 391     return _FALSE;
 392 }

 394 static _BOOL stHdlConnect_()
 395 {
 396     static _UWORD counter = SSRP_PACKET_SPC;
 397     _UBYTE buf = 0;
 398     _SINT i;
 399 
 400     if ((trx_[SSRP_TRX_RECV].header.to == my_addr_) &&
 401         SSRP_COMMAND_EQ(&trx_[SSRP_TRX_RECV].header,
 402                         SSRP_CMD_JOIN, SSRP_FLG_RES))
 403     {
 404         insertNode_(trx_[SSRP_TRX_RECV].header.from);
 405 
 406         pear_.con[*(trx_[SSRP_TRX_RECV].buffer)] =
 407             trx_[SSRP_TRX_RECV].header.from;
 408     }
 409     else
 410     {
 411         sendResponse_();
 412     }
 413 
 414     if (pear_.addr_count == pear_.total)
 415     {
 416         for (i = 0; i < SSRP_NUM_OF_CON; i++)
 417         {
 418             if (pear_.con[i] == SSRP_ADDR_INVALID)
 419             {
 420                 goto func_END;
 421             }
 422         }
 423         return _TRUE;
 424     }
 425 
 426 func_END:
 427     if (counter++ & SSRP_PACKET_SPC)
 428     {
 429         SSRP_sendto(SSRP_ADDR_BROADCAST, SSRP_CMD_JOIN, SSRP_FLG_REQ,
 430                     &buf, sizeof(buf), NULL, _TRUE);
 431     }
 432 
 433     return _FALSE;
 434 }

 458 }

 460 static void sendResponse_()
 461 {
 462     _UBYTE buf;
 463     _SINT rsize;
 464 
 465     if (SSRP_IS_INVALID_PACKET(&trx_[SSRP_TRX_RECV].header)) return;
 466     if (trx_[SSRP_TRX_RECV].header.from == my_addr_) return;
 467 
 468     if (SSRP_COMMAND_EQ(&trx_[SSRP_TRX_RECV].header,
 469                         SSRP_CMD_JOIN, SSRP_FLG_REQ))
 470     {
 471         insertNode_(trx_[SSRP_TRX_RECV].header.from);
 472 
 473         if (trx_[SSRP_TRX_RECV].header.hop == 0)
 474         {
 475             buf = SSRP_CON_FRONT;
 476             rsize = SSRP_sendto(trx_[SSRP_TRX_RECV].header.from,
 477                                 SSRP_CMD_JOIN, SSRP_FLG_RES,
 478                                 &buf, sizeof(buf), NULL, _TRUE);
 479             if (rsize > 0)
 480             {
 481                 pear_.con[SSRP_CON_BACK] =
 482                     trx_[SSRP_TRX_RECV].header.from;
 483             }
 484         }
 485 
 486         if (trx_[SSRP_TRX_RECV].header.hop == (pear_.total - 1))
 487         {
 488             buf = SSRP_CON_BACK;
 489             rsize = SSRP_sendto(trx_[SSRP_TRX_RECV].header.from,
 490                                 SSRP_CMD_JOIN, SSRP_FLG_RES,
 491                                 &buf, sizeof(buf), NULL, _TRUE);
 492             if (rsize > 0)
 493             {
 494                 pear_.con[SSRP_CON_FRONT] = trx_[SSRP_TRX_RECV].header.from;
 495             }
 496         }
 497     }
 498     else if (SSRP_COMMAND_EQ(&trx_[SSRP_TRX_RECV].header,
 499                              SSRP_CMD_LEAVE, SSRP_FLG_EVT))
 500     {
 501         _SINT i;
 502 
 503         for (i = 0; i < SSRP_NUM_OF_CON; i++)
 504         {
 505             if (pear_.con[i] == trx_[SSRP_TRX_RECV].header.from)
 506             {
 507                 pear_.con[i] = SSRP_ADDR_INVALID;
 508             }
 509         }
 510 
 511         deleteNode_(trx_[SSRP_TRX_RECV].header.from);
 512     }
 513 }

 515 static _SWORD findNode_(SSRP_Address addr)
 516 {
 517     _SWORD rtn = -1;
 518     _SWORD idx = _hash(addr, SSRP_MAX_NODE);
 519     _SINT i = SSRP_MAX_NODE;
 520     
 521     while (i--)
 522     {
 523         if (pear_.addr[idx] == addr)
 524         {
 525             rtn = idx;
 526             break;
 527         }
 528         idx++;
 529     }
 530 
 531     return rtn;
 532 }

 534 static _BOOL insertNode_(SSRP_Address addr)
 535 {
 536     _BOOL rtn = _FALSE;
 537     _SWORD idx = _hash(addr, SSRP_MAX_NODE);
 538     _SINT i = SSRP_MAX_NODE;
 539     
 540     while (i--)
 541     {
 542         if (pear_.addr[idx] == SSRP_ADDR_INVALID)
 543         {
 544             pear_.addr[idx] = addr;
 545             pear_.addr_count++;
 546             rtn = _TRUE;
 547             break;
 548         }
 549         else if (pear_.addr[idx] == addr)
 550         {
 551             break;
 552         }
 553         idx++;
 554     }
 555 
 556     return rtn;
 557 }

 559 static _BOOL deleteNode_(SSRP_Address addr)
 560 {
 561     _BOOL rtn = _FALSE;
 562     _SWORD idx = _hash(addr, SSRP_MAX_NODE);
 563     _SINT i = SSRP_MAX_NODE;
 564     
 565     while (i--)
 566     {
 567         if (pear_.addr[idx] == addr)
 568         {
 569             pear_.addr[idx] = SSRP_ADDR_INVALID;
 570             --pear_.addr_count;
 571             rtn = _TRUE;
 572             break;
 573         }
 574         idx++;
 575     }
 576 
 577     return rtn;
