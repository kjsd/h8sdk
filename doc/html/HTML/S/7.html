 135 _BOOL MUSIC_reverse()
 136 {
 137     _SINT i, j;
 138 
 139     if (!state_mtx_[current_state_][MUSIC_ST_REVERSE]) return _FALSE;
 140 
 141     _assert(score_);
 142 
 143     SOUND_stop();
 144 
 145     if (current_state_ == MUSIC_ST_STOP)
 146     {
 147         for (i = 0; i < score_->size; i++)
 148         {
 149             position_.note[i] = PART_(i).size - 1;
 150             if (position_.count < (NOTE_(i).begin + NOTE_(i).len))
 151             {
 152                 position_.count = (NOTE_(i).begin + NOTE_(i).len);
 153             }
 154         }
 155     }
 156 
 157     current_state_ = MUSIC_ST_REVERSE;
 158     SOUND_reverse();
 159 
 160     return _TRUE;
 161 }

 173 _BOOL MUSIC_stop()
 174 {
 175     _SINT i;
 176 
 177     if (!state_mtx_[current_state_][MUSIC_ST_STOP]) return _FALSE;
 178 
 179     SOUND_stop();
 180 
 181     for (i = 0; i < score_->size; i++)
 182     {
 183         SOUND_removeTone(i);
 184     }
 185 
 186     part_finish_ = 0;
 187     part_pronounce_ = 0;
 188     _memset(&position_, 0, sizeof(position_));
 189     current_state_ = MUSIC_ST_STOP;
 190 
 191     return _TRUE;
 192 }

 199 static _BOOL playFinalize_(MUSIC_State prev)
 200 {
 201     if (part_finish_ == ((1U << score_->size) - 1))
 202     {
 203         MUSIC_stop();
 204         if (loop_)
 205         {
 206             MUSIC_state_handler[prev]();
 207         }
 208 
 209         return _TRUE;
 210     }
 211     return _FALSE;
 212 }

 219 static void stPlayHandle_()
 220 {
 221     _SINT i;
 222 
 223     if (playFinalize_(MUSIC_ST_PLAY)) return;
 224 
 225     for (i = 0; i < score_->size; i++)
 226     {
 227         if (part_finish_ & (1U << i)) continue;
 228 
 229         if (part_pronounce_ & (1U << i))
 230         {
 231             _BOOL tone_change = _FALSE;
 232             while (position_.count > (NOTE_(i).begin + NOTE_(i).len))
 233             {
 234                 if (++position_.note[i] == PART_(i).size)
 235                 {
 236                     part_finish_ |= (1U << i);
 237                     SOUND_removeTone(i);
 238                     tone_change = _FALSE;
 239                     break;
 240                 }
 241                 tone_change = _TRUE;
 242             }
 243             if (!tone_change) continue;
 244 
 245             if (position_.count >= NOTE_(i).begin)
 246             {
 247                 if (NOTE_(i).tone.freq == SOUND_NA)
 248                 {
 249                     SOUND_removeTone(i);
 250                 }
 251                 else
 252                 {
 253                     SOUND_setTone(i, &NOTE_(i).tone);
 254                 }
 255             }
 256             else
 257             {
 258                 SOUND_removeTone(i);
 259                 part_pronounce_ &= ~(1U << i);
 260             }
 261         }
 262         else
 263         {
 264             if (position_.count >= NOTE_(i).begin)
 265             {
 266                 if (NOTE_(i).tone.freq != SOUND_NA)
 267                 {
 268                     SOUND_setTone(i, &NOTE_(i).tone);
 269                 }
 270                 part_pronounce_ |= (1U << i);
 271             }
 272         }
 273     }
 274 
 275     position_.count++;
 276 }

 278 static void stReverseHandle_()
 279 {
 280     _SINT i;
 281 
 282     if (playFinalize_(MUSIC_ST_REVERSE)) return;
 283 
 284     for (i = 0; i < score_->size; i++)
 285     {
 286         if (part_finish_ & (1U << i)) continue;
 287 
 288         if (part_pronounce_ & (1U << i))
 289         {
 290             _BOOL tone_change = _FALSE;
 291             while (position_.count <= NOTE_(i).begin)
 292             {
 293                 if (position_.note[i]-- == 0)
 294                 {
 295                     part_finish_ |= (1U << i);
 296                     SOUND_removeTone(i);
 297                     tone_change = _FALSE;
 298                     break;
 299                 }
 300                 tone_change = _TRUE;
 301             }
 302             if (!tone_change) continue;
 303 
 304             if (position_.count <= (NOTE_(i).begin + NOTE_(i).len))
 305             {
 306                 if (NOTE_(i).tone.freq == SOUND_NA)
 307                 {
 308                     SOUND_removeTone(i);
 309                 }
 310                 else
 311                 {
 312                     SOUND_setTone(i, &NOTE_(i).tone);
 313                 }
 314             }
 315             else
 316             {
 317                 SOUND_removeTone(i);
 318                 part_pronounce_ &= ~(1U << i);
 319             }
 320         }
 321         else
 322         {
 323             if (position_.count <= (NOTE_(i).begin + NOTE_(i).len))
 324             {
 325                 if (NOTE_(i).tone.freq != SOUND_NA)
 326                 {
 327                     SOUND_setTone(i, &NOTE_(i).tone);
 328                 }
 329                 part_pronounce_ |= (1U << i);
 330             }
 331         }
 332     }
 333 
 334     --position_.count;
 335 }

 342 #ifdef USE_MUSIC_RECORD
 343 _BOOL MUSIC_record()
 344 {
 345     _SINT i;
 346     SOUND_State st;
 347 
 348     if (!state_mtx_[current_state_][MUSIC_ST_RECORD]) return _FALSE;
 349 
 350     st = SOUND_getState();
 351     if (st != SOUND_ST_STOP)
 352     {
 353         SOUND_stop();
 354     }
 355 
 356     rec_score_.size = 0;
 357 
 358     for (i = 0; i < _sizeof_array(rec_parts_); i++)
 359     {
 360         rec_parts_[i].notes = rec_notes_[i];
 361         rec_parts_[i].size = 0;
 362     }
 363     _memset(rec_notes_, 0, sizeof(rec_notes_));
 364     _memset(rec_param_, 0, sizeof(rec_param_));
 365     _memset(&rec_position_, 0, sizeof(rec_position_));
 366     rec_part_ = 0;
 367 
 368     _memset(&position_, 0, sizeof(position_));
 369     score_ = &rec_score_;
 370 
 371     current_state_ = MUSIC_ST_RECORD;
 372 
 373     if (st != SOUND_ST_STOP)
 374     {
 375         SOUND_state_handler[st]();
 376     }
 377     else
 378     {
 379         SOUND_play();
 380     }
 381 
 382     return _TRUE;
 383 }

 410 static MUSIC_RecordReqId newRecordNote_(const SOUND_Tone* tone)
 411 {
 412     static _UWORD req_id = 1;
 413     _SINT i;
 414     _UBYTE idx;
 415     _UWORD size = _sizeof_array(rec_param_);
 416 
 417     idx = _hash(req_id, size);
 418     while (size--)
 419     {
 420         if (rec_param_[idx].reqid == MUSIC_RECREQID_INVALID)
 421         {
 422             if (rec_position_.note[rec_part_] == MUSIC_NOTE_OF_RECORD_PART)
 423             {
 424                 return MUSIC_RECREQID_INVALID;
 425             }
 426 
 427             rec_param_[idx].reqid = req_id;
 428             rec_param_[idx].part_idx = rec_part_;
 429             rec_param_[idx].note_idx = rec_position_.
 430                 note[rec_param_[idx].part_idx];
 431             rec_notes_[rec_param_[idx].part_idx][rec_param_[idx].note_idx].
 432                 begin = rec_position_.count;
 433 
 434             _memcpy(&rec_notes_[rec_param_[idx].part_idx]
 435                     [rec_param_[idx].note_idx].tone,
 436                     tone,
 437                     sizeof(SOUND_Tone));
 438 
 439             rec_part_++;
 440             rec_position_.note[rec_param_[idx].part_idx]++;
 441 
 442             rec_parts_[rec_param_[idx].part_idx].size++;
 443             if (rec_score_.size < rec_part_)
 444             {
 445                 rec_score_.size = rec_part_;
 446             }
 447 
 448             if (++req_id == MUSIC_RECREQID_INVALID)
 449             {
 450                 req_id++;
 451             }
 452             return rec_param_[idx].reqid;
 453         }
 454 
 455         idx = _next_ring(idx, _sizeof_array(rec_param_) - 1);
 456     }
 457 
 458     return MUSIC_RECREQID_INVALID;
 459 }

 461 static MUSIC_Note* deleteRecordNote_(MUSIC_RecordReqId key)
 462 {
 463     _UBYTE idx;
 464     _UWORD size = _sizeof_array(rec_param_);
 465 
 466     idx = _hash(key, size);
 467     while (size--)
 468     {
 469         if (rec_param_[idx].reqid == key)
 470         {
 471             rec_param_[idx].reqid = MUSIC_RECREQID_INVALID;
 472             --rec_part_;
 473 
 474             return &rec_notes_[rec_param_[idx].part_idx]
 475                 [rec_param_[idx].note_idx];
 476         }
 477 
 478         idx = _next_ring(idx, _sizeof_array(rec_param_) - 1);
 479     }
 480 
 481     return NULL;
 482 }

 484 static void stRecordHandle_()
 485 {
 486     _SINT i;
 487 
 488     rec_position_.count++;
 489 
 490     for (i = 0; i < _sizeof_array(rec_param_); i++)
 491     {
 492         if (rec_param_[i].reqid != MUSIC_RECREQID_INVALID)
 493         {
 494             rec_notes_[rec_param_[i].part_idx][rec_param_[i].note_idx].len++;
 495         }
 496     }
