 101 void SOUND_init()
 102 {
 103     /* TimerW */
 104     IOCTL_set(IOCTL_REG_PCR8, 0x08, 0x08);
 105     _IO.PDR8.BIT.B3 = 0;
 106     _TW.TMRW.BIT.CTS = 0;
 107     _TW.TMRW.BIT.PWMC = 0;
 108     _TW.TCRW.BIT.CCLR = 1;
 109     _TW.TCRW.BIT.CKS = 0;
 110     _TW.TCRW.BIT.TOC = 1;
 111     _TW.TIOR1.BYTE = 0;
 112     _TW.GRA = 0x100;
 113 
 114     /* TimerV */
 115     _TV.TCSRV.BYTE = 0x06;
 116     _TV.TCRV1.BYTE = 0xe3;
 117     _TV.TCRV0.BYTE = 0x0b;
 118 }

 138 void SOUND_setTone(_UBYTE idx, const SOUND_Tone* tone)
 139 {
 140     _assert(idx < SOUND_MAX_PRONOUNCE);
 141     _assert(tone != NULL);
 142     _assert((tone->freq != SOUND_NA) && (tone->evlp != NULL));
 143 
 144     STOP_TIMER_;
 145 
 146     if (tones_[idx].freq == SOUND_NA)
 147     {
 148         num_of_tones_++;
 149     }
 150     tones_[idx] = *tone;
 151     is_reverb_[idx] = _FALSE;
 152     remove_tone_[idx] = _FALSE;
 153 
 154     if (num_of_tones_ == 1)
 155     {
 156         tone_pos_ = idx;
 157     }
 158 
 159     switch (current_state_)
 160     {
 161     case SOUND_ST_PLAY:
 162         evlp_pos_[idx] = 0;
 163         if (num_of_tones_ == 1)
 164         {
 165             evlp_ = tones_[tone_pos_].evlp->data[evlp_pos_[tone_pos_]] *
 166                 volume_;
 167         }
 168 
 169         CLEAR_TIMER_;
 170         START_TIMER_;
 171         break;
 172 
 173     case SOUND_ST_REVERSE:
 174         evlp_pos_[idx] = tones_[idx].evlp->size - 1;
 175         if (num_of_tones_ == 1)
 176         {
 177             evlp_ = tones_[tone_pos_].evlp->data[evlp_pos_[tone_pos_]] *
 178                 volume_;
 179         }
 180 
 181         CLEAR_TIMER_;
 182         START_TIMER_;
 183         break;
 184 
 185     case SOUND_ST_STOP:
 186     case SOUND_NUM_OF_STATE:
 187     default:
 188         break;
 189     }
 190 }

 192 void SOUND_removeTone(_UBYTE idx)
 193 {
 194     _SINT i;
 195 
 196     _assert(idx < SOUND_MAX_PRONOUNCE);
 197 
 198     if (tones_[idx].freq == SOUND_NA)
 199     {
 200         return;
 201     }
 202 
 203     switch (current_state_)
 204     {
 205     case SOUND_ST_PLAY:
 206         /* fallthrough */
 207     case SOUND_ST_REVERSE:
 208         if (!is_reverb_[idx] && (tones_[idx].reverb != NULL))
 209         {
 210             evlp_pos_[idx] = 0;
 211             is_reverb_[idx] = _TRUE;
 212         }
 213 
 214         remove_tone_[idx] = _TRUE;
 215         break;
 216 
 217     case SOUND_ST_STOP:
 218     case SOUND_NUM_OF_STATE:
 219     default:
 220         deleteTone_(idx);
 221         break;
 222     }
 223 }

 265 void SOUND_makePulse()
 266 {
 267     if (!IS_PLAYING_STATE_(current_state_) || (num_of_tones_ == 0))
 268     {
 269         _TV.TCSRV.BIT.CMFA = 0;
 270         _TV.TCSRV.BIT.CMFB = 0;
 271         return;
 272     }
 273 
 274     if (_TV.TCSRV.BIT.CMFA)
 275     {
 276         _TV.TCSRV.BIT.CMFA = 0;
 277         _TW.TMRW.BIT.PWMC = evlp_ ? 1: 0;
 278     }
 279     if (_TV.TCSRV.BIT.CMFB)
 280     {
 281         _TV.TCSRV.BIT.CMFB = 0;
 282         _TW.TMRW.BIT.PWMC = 0;
 283     }
 284 }

 286 void SOUND_pronounce()
 287 {
 288     _UWORD tcnt;
 289     _BOOL tone_change = _FALSE;
 290 
 291     if (!IS_PLAYING_STATE_(current_state_) || (num_of_tones_ == 0))
 292     {
 293         return;
 294     }
 295 
 296     if (num_of_tones_ > 1)
 297     {
 298         if (++pronounce_cnt_ == SOUND_PRONOUNCE_LEN)
 299         {
 300             _SINT i;
 301             for (i = 0; i < SOUND_MAX_PRONOUNCE; i++)
 302             {
 303                 tone_pos_ = _next_ring(tone_pos_, SOUND_MAX_PRONOUNCE - 1);
 304                 if (tones_[tone_pos_].evlp != NULL)
 305                 {
 306                     tone_change = _TRUE;
 307                     break;
 308                 }
 309             }
 310             pronounce_cnt_ = 0;
 311         }
 312     }
 313 
 314     setNextEvlp_();
 315 
 316     if (tone_change)
 317     {
 318         _TW.TMRW.BIT.PWMC = 0;
 319         CLEAR_TIMER_;
 320         _TW.TMRW.BIT.PWMC = 1;
 321     }
 322     else
 323     {
 324         if (_TW.GRA >= 0x100)
 325         {
 326             tcnt = (_TW.GRC < _TW.GRA) ? _TW.GRC: _TW.GRA;
 327             while (_TW.TCNT < tcnt);
 328         }
 329         _TW.GRC = evlp_;
 330     }
 331 }

 333 static void setNextEvlp_()
 334 {
 335     SOUND_Envelope* p;
 336 
 337     p = (is_reverb_[tone_pos_]) ?
 338         tones_[tone_pos_].reverb: tones_[tone_pos_].evlp;
 339 
 340     if (p == NULL)
 341     {
 342         evlp_ = 0;
 343         return;
 344     }
 345 
 346     switch (current_state_)
 347     {
 348     case SOUND_ST_PLAY:
 349         if (evlp_pos_[tone_pos_] < p->size)
 350         {
 351             evlp_ = p->data[evlp_pos_[tone_pos_]] * volume_;
 352             evlp_pos_[tone_pos_] += num_of_tones_;
 353             return;
 354         }
 355 
 356         if (!is_reverb_[tone_pos_] && (tones_[tone_pos_].reverb != NULL))
 357         {
 358             evlp_pos_[tone_pos_] = 0;
 359             is_reverb_[tone_pos_] = _TRUE;
 360         }
 361         else
 362         {
 363             if (tones_[tone_pos_].loop && !remove_tone_[tone_pos_])
 364             {
 365                 evlp_pos_[tone_pos_] = 0;
 366             }
 367             else
 368             {
 369                 evlp_ = 0;
 370                 deleteTone_(tone_pos_);
 371             }
 372         }
 373         break;
 374 
 375     case SOUND_ST_REVERSE:
 376         if (is_reverb_[tone_pos_])
 377         {
 378             if (evlp_pos_[tone_pos_] < p->size)
 379             {
 380                 evlp_ = p->data[evlp_pos_[tone_pos_]] * volume_;
 381                 evlp_pos_[tone_pos_] += num_of_tones_;
 382                 return;
 383             }
 384         }
 385         else
 386         {
 387             if (evlp_pos_[tone_pos_] >= num_of_tones_)
 388             {
 389                 evlp_ = p->data[evlp_pos_[tone_pos_]] * volume_;
 390                 evlp_pos_[tone_pos_] -= num_of_tones_;
 391                 return;
 392             }
 393         }
 394 
 395         if (!is_reverb_[tone_pos_] && (tones_[tone_pos_].reverb != NULL))
 396         {
 397             evlp_pos_[tone_pos_] = 0;
 398             is_reverb_[tone_pos_] = _TRUE;
 399         }
 400         else
 401         {
 402             if (tones_[tone_pos_].loop && !remove_tone_[tone_pos_])
 403             {
 404                 evlp_pos_[tone_pos_] = tones_[tone_pos_].evlp->size - 1;
 405             }
 406             else
 407             {
 408                 evlp_ = 0;
 409                 deleteTone_(tone_pos_);
 410             }
 411         }
 412         break;
 413 
 414     case SOUND_ST_STOP:
 415     case SOUND_NUM_OF_STATE:
 416     default:
 417         _assert(_FALSE);
 418         break;
 419     }
 420 }

 422 static void deleteTone_(_UBYTE idx)
 423 {
 424     _assert(idx < SOUND_MAX_PRONOUNCE);
 425     _assert((tones_[idx].freq != SOUND_NA) &&
 426                 (tones_[idx].evlp != NULL));
 427 
 428     --num_of_tones_;
 429 
 430     tones_[idx].freq = SOUND_NA;
 431     tones_[idx].evlp = NULL;
 432     tones_[idx].reverb = NULL;
 433     tones_[idx].loop = _FALSE;
 434     evlp_ = 0;
 435 
 436     if ((num_of_tones_ > 0) && (tone_pos_ == idx))
 437     {
 438         _SINT i;
 439 
 440         for (i = 0; i < SOUND_MAX_PRONOUNCE; i++)
 441         {
 442             if (tones_[i].evlp != NULL)
 443             {
 444                 tone_pos_ = i;
 445                 CLEAR_TIMER_;
 446                 break;
 447             }
 448         }
 449     }
